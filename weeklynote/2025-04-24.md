```
function grab_pkgs_from_markdown() {
	local MD_FPATH="$1"

	python3 -c "$(
		cat <<EOPY
import markdown_it

md_text = open("$MD_FPATH", encoding='utf-8').read()
tokens = markdown_it.MarkdownIt().parse(md_text)

for token in tokens:
    if token.type == "fence" and token.info.strip() == "archup":
        lines = token.content.strip()
        for line in lines.split("\n"):
            print(line.strip())
EOPY
	)"

}
```
